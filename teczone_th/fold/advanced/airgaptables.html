ตารางระยะช่องว่างอากาศแบบกำหนดเอง

ตารางระยะช่องว่างอากาศสำหรับเครื่องดัดแผงจะทำการจับคู่ความหนาของแผ่นโลหะกับค่าระยะช่องว่างอากาศมาตรฐาน ที่ใช้ระหว่างกระบวนการ โดยจะกำหนดไว้ในเครื่องดัดแผงแต่ละเครื่องอยู่แล้ว แต่สามารถเขียนทับได้ด้วยการระบุไฟล์ตารางระยะช่องว่างอากาศแบบกำหนดเองในรูปแบบไฟล์ JSON ไฟล์ JSON นี้มีชื่อว่า air-gap-tables.json และถูกจัดเก็บไว้ที่ C:\ProgramData\Trumpf{APPNAME}\Data\System

ด้านล่างนี้เป็นตัวอย่างไฟล์ air-gap-tables.json ทั่วไป:

 {
  "version":1  "airGapTables": [
    {
      "machine": "112002,112003", "material": "DC01"
      "values": [
        { "t": 0.75, "gap": 2.5 },
        { "t": 0.95, "gap": 3 },
        { "t": 1.25, "gap": 3.5 },
        { "t": 1.45, "gap": 4.5 },
        { "t": 1.65, "gap": 5.5 },
        { "t": 1.75, "gap": 6 },
        { "t": 2.05, "gap": 6.5 },
        { "t": 2.15, "gap": 7.5 },
        { "t": 2.55, "gap": 8 },
        { "t": 9.99, "gap": 8.5 }
      ]
    },
     {
      "machine": "112003", "material": "*"
      "values": [
        { "t": 0.75, "gap": 2.5 },
        { "t": 0.95, "gap": 3 },
        { "t": 1.25, "gap": 3.5 },
        { "t": 1.45, "gap": 4.5 },
        { "t": 1.65, "gap": 5.5 },
        { "t": 1.75, "gap": 6 },
        { "t": 2.05, "gap": 6.5 },
        { "t": 2.15, "gap": 7.5 },
        { "t": 2.55, "gap": 8 },
        { "t": 9.99, "gap": 8.5 }
      ]
    },
    {
      "machine": "*", "material": "Alluminio"
      "values": [
        { "t": 0.75, "gap": 2.5 },
        { "t": 0.95, "gap": 3 },
        { "t": 1.25, "gap": 3.5 },
        { "t": 1.45, "gap": 4.5 },
        { "t": 1.65, "gap": 5.5 },
        { "t": 1.75, "gap": 6 },
        { "t": 2.05, "gap": 6.5 },
        { "t": 2.15, "gap": 7.5 },
        { "t": 2.55, "gap": 8 },
        { "t": 9.99, "gap": 8.5 }
      ]
    }
  ]
}

  • ไฟล์นี้ประกอบด้วยชุดของค่า air-gap แต่ละตารางจะมี machine key และ material key ซึ่งใช้เป็นตัวเลือก

  • คีย์เหล่านี้จะถูกกำหนดเป็น * (ดอกจัน) เพื่อระบุว่าเป็น wildcard (ตรงกับทุกค่า) หรือกำหนดเป็นรายการของค่าเครื่องจักรหรือวัสดุ โดยคั่นด้วยเครื่องหมายจุลภาค

  • ในตัวอย่างข้างต้น มีอยู่สามตาราง: ตารางแรกใช้สำหรับเครื่องจักร 112001 หรือ 112003 โดยเฉพาะวัสดุ DC01 ตารางที่สองใช้สำหรับวัสดุทั้งหมด แต่เฉพาะเครื่องจักร 112003 ตารางที่สามใช้สำหรับเครื่องจักรทั้งหมด แต่เฉพาะวัสดุ Alluminio หากมีหลายตารางที่ตรงกับการจับคู่เครื่องจักร/วัสดุของชิ้นงาน ตารางแรก ที่ตรงกันจะถูกนำมาใช้ ดังนั้น สำหรับเครื่องจักร 112003 วัสดุ DC01 จะใช้ตารางแรก (แม้ว่าตารางที่สองจะตรงด้วยก็ตาม)

  • แต่ละตารางประกอบด้วยอาร์เรย์ของ ค่าความหนาพร้อมด้วยค่าของ air-gap ที่สอดคล้องกัน ตัวอย่างเช่น รายการแรกด้านบนควรถูกตีความว่า: หากความหนาน้อยกว่าหรือเท่ากับ 0.75 มม. ให้ใช้ค่า gap เท่ากับ 2.5 มม. ปุ่มความหนาจะต้องเรียง ตามลำดับจากน้อยไปมาก หากไม่มีตารางใดในไฟล์นี้ที่ตรงกับการจับคู่ของวัสดุและความหนาของชิ้นงาน จะมีการใช้ตาราง air-gap ค่าเริ่มต้นของเครื่องจักร

หากคุณมีการเปลี่ยนแปลงใดๆ ในไฟล์ air-gap-tables.json คุณจำเป็นต้องรีสตาร์ท TecZone Fold เพื่อให้การเปลี่ยนแปลงนั้นมีผล
หากไฟล์ JSON มีรูปแบบไม่ถูกต้อง ระบบจะแสดงข้อความ error เมื่อคุณพยายามประมวลผลชิ้นงานของเครื่องดัดแผง แต่จากนั้น TecZone Fold จะย้อนกลับไปในตาราง air-gap ค่าเริ่มต้นของเครื่องจักร และดำเนินการต่อไป
ขั้นสูง การทำงานของการดัดแบบแมนนวล
